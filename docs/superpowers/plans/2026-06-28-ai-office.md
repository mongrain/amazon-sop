# AI 办公室 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在 sop-system 中新增 AI 办公室模块：5 名 AI 员工、手动派活、指派即执行、主管必审、老板自动拆单分派；Phase 1 为功能看板，Phase 2 为 2D 可视化。

**Architecture:** 独立表 `ai_agents` / `ai_office_tasks` / `ai_office_task_logs`，业务逻辑在 `service/ai-office.js`，异步执行在 `orchestration/ai-office/`（仿 `orchestration/product-selection/`），GPT 复用 `gpt.js`。前端新增两个 Vue 页面，侧边栏加入口。

**Tech Stack:** Node.js Express, MySQL (Sequelize raw SQL), Vue 3 + Vue Router, gpt.js (OpenAI-compatible)

**Spec:** `docs/superpowers/specs/2026-06-28-ai-office-design.md`

        try {
            const task = await aiOffice.createTask({
                title: req.body.title,
                description: req.body.description,
                assigned_agent_code: req.body.assigned_agent_code,
                priority: req.body.priority || 'NORMAL',
                context_json: req.body.context_json || { source: 'manual' },
                created_by: req.currentUser.id
            });
            dispatchAiOfficeTask(task.id);
            res.json({ task });
        } catch (e) {
            res.status(400).json({ error: e.message });
        }
    });

    app.get('/api/ai-office/tasks/:id', async (req, res) => {
        try {
            const data = await aiOffice.getTaskDetail(req.params.id);
            if (!data) return res.status(404).json({ error: '任务不存在' });
            res.json(data);
        } catch (e) {
            res.status(500).json({ error: e.message });
        }

- [ ] **Step 1: dispatch（仿 product-selection）**

```javascript
const { queryOne, runSql } = require('../../database');
const aiOffice = require('../../service/ai-office');
const { runAgentTask } = require('./runner');

function dispatchAiOfficeTask(taskId) {
    setImmediate(() => executeAiOfficeTask(taskId));
}

async function executeAiOfficeTask(taskId) {
    try {
        await aiOffice.updateTaskStatus(taskId, 'IN_PROGRESS');
        await aiOffice.refreshAgentStatuses();
        const detail = await aiOffice.getTaskDetail(taskId);
        if (!detail || !detail.task) return;

        const agent = detail.agent;
        if (!agent) throw new Error('未指派员工');

        await runAgentTask(detail.task, agent);

        // runner 内部处理 boss 拆单；普通任务完成后进入 PENDING_REVIEW 并 dispatch 主管审核
    } catch (e) {
        await aiOffice.updateTaskStatus(taskId, 'FAILED', { error_message: String(e.message || e) });
        await aiOffice.appendLog(taskId, { log_type: 'system', content: '执行失败: ' + e.message });
    } finally {
        await aiOffice.refreshAgentStatuses();
    }
}

module.exports = { dispatchAiOfficeTask, executeAiOfficeTask };
```

- [ ] **Step 2: runner 路由到各 agent 文件**

```javascript
const boss = require('./agents/boss');
const designer = require('./agents/designer');
const analyst = require('./agents/analyst');
const researcher = require('./agents/researcher');
const supervisor = require('./agents/supervisor');
const aiOffice = require('../../service/ai-office');
const { dispatchAiOfficeTask } = require('./index');

async function runAgentTask(task, agent) {
    const handlers = { boss, designer, analyst, researcher };
    const handler = handlers[agent.code];
    if (!handler) throw new Error('未知员工: ' + agent.code);

    const output = await handler.run(task, agent);

    if (agent.code === 'boss') {
        // boss.run 内部已创建子任务并 dispatch，父任务保持 IN_PROGRESS 直到子任务全 DONE
        await checkParentCompletion(task.id);
        return;
    }

    await aiOffice.updateTaskStatus(task.id, 'PENDING_REVIEW', { output_markdown: output });
    await aiOffice.appendLog(task.id, { agent_id: agent.id, log_type: 'agent', content: output });
    await runSupervisorReview(task.id);
}

async function runSupervisorReview(taskId) {
    const detail = await aiOffice.getTaskDetail(taskId);
    const result = await supervisor.review(detail.task, detail.task.output_markdown);
    if (result.approved) {
        await aiOffice.updateTaskStatus(taskId, 'DONE', { review_comment: result.comment, completed_at: new Date() });
        await checkParentCompletion(detail.task.parent_task_id);
    } else {
        const retry = (detail.task.retry_count || 0) + 1;
        if (retry >= aiOffice.MAX_RETRIES) {
            await aiOffice.updateTaskStatus(taskId, 'FAILED', { error_message: '审核打回次数超限', review_comment: result.comment });
        } else {
            await runSql('UPDATE ai_office_tasks SET retry_count = ?, review_comment = ? WHERE id = ?', [retry, result.comment, taskId]);
            await aiOffice.appendLog(taskId, { log_type: 'review', content: '打回: ' + result.comment });
            dispatchAiOfficeTask(taskId);
        }
    }
}

async function checkParentCompletion(parentId) {
    if (!parentId) return;
    const subs = await aiOffice.listTasks({ parent_task_id: parentId });
    if (subs.length && subs.every(t => t.status === 'DONE')) {
        await aiOffice.updateTaskStatus(parentId, 'DONE', { completed_at: new Date() });
    }
}

module.exports = { runAgentTask, runSupervisorReview };
```
